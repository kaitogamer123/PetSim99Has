-- УЛЬТИМАТИВНЫЙ МЕНЕДЖЕР: АВТО-АПГРЕЙД (АКТУАЛЬНАЯ ВЕРСИЯ)
-- ==========================================================

_G.AutoEasterManager = false
task.wait(0.5)
_G.AutoEasterManager = true

task.spawn(function()
    -- Подключаем игровые библиотеки из твоего исходника
    local Lib = game:GetService("ReplicatedStorage"):WaitForChild("Library")
    local EventUpgradeCmds = require(Lib:WaitForChild("Client"):WaitForChild("EventUpgradeCmds"))
    local EventUpgradeDir = require(Lib:WaitForChild("Directory"):WaitForChild("EventUpgrades"))
    local Save = require(Lib:WaitForChild("Client"):WaitForChild("Save"))
    local EasterTypes = require(Lib:WaitForChild("Types"):WaitForChild("Easter2026"))

    -- Треки из исходника игры
    local baseTracks = {"Luck", "ShinyLuck", "HatchSpeed", "CooldownAndAmount"}

    -- Функция определения актуальных ID (как в игре)
    local function getCurrentUpgradeIds()
        local data = Save.Get()
        local activeEgg = data and data.Easter2026ActiveEgg or 1
        local ids = {}
        
        -- Базовые апгрейды
        for _, track in ipairs(baseTracks) do
            table.insert(ids, "Easter2026Egg" .. activeEgg .. track)
        end
        
        -- Апгрейды Босс-сундука (BossChestTracks)
        local bossTracks = EasterTypes.GetBossChestTracks(activeEgg)
        for _, track in ipairs(bossTracks) do
            table.insert(ids, "Easter2026Egg" .. activeEgg .. track)
        end
        
        return ids
    end

    print("--- [MANAGER]: Скрипт активен. Только апгрейды. ---")

    while _G.AutoEasterManager do
        pcall(function()
            local currentIds = getCurrentUpgradeIds()

            for _, upgradeId in ipairs(currentIds) do
                local upgradeData = EventUpgradeDir[upgradeId]
                
                if upgradeData then
                    -- Проверяем, можно ли купить (хватает ли ресурсов и не макс. ли уровень)
                    local currentTier = EventUpgradeCmds.GetTier(upgradeData)
                    local maxTier = #upgradeData.TierPowers

                    if currentTier < maxTier then
                        -- Пытаемся купить, пока сервер разрешает
                        local bought
                        repeat
                            bought = EventUpgradeCmds.Purchase(upgradeData)
                            if bought then
                                print("🚀 [UPGRADE]: Куплен " .. upgradeId .. " (Уровень " .. (currentTier + 1) .. ")")
                                task.wait(0.2)
                            end
                        until not bought or not _G.AutoEasterManager
                    end
                end
            end
        end)

        task.wait(30) -- Проверка каждые 30 секунд
    end
    print("--- [MANAGER]: Скрипт остановлен. ---")
end)
